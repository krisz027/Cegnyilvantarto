401         /*HOZZAAD GOMB*/
402         
403         hibasadat = 0;
404             /*adatok vizsgálata*/   
405         if(jTextField1.getText().isEmpty()){
406             segedpeldany.hibasBevitelFigyelmeztetes(jTextField1, "Nincs megadva az ország!");
407         }
408         if(hibasadat == 0){
409             if(jTextField2.getText().isEmpty()){
410                 segedpeldany.hibasBevitelFigyelmeztetes(jTextField2, "Nincs megadva irányítószám!");
411             }
412         }
413         if(hibasadat == 0){
414             for(int i=0; i<jTextField2.getText().length(); i++){
415                 if(!Character.isDigit(jTextField2.getText().charAt(i))){
416                     segedpeldany.hibasBevitelFigyelmeztetes(jTextField2, "Helytelen irányítószám!");
417                     break;
418                 }
419             }
420         }
421         if(hibasadat == 0){
422             for(int i=0; i<jTextField5.getText().length(); i++){
423                 if(!Character.isDigit(jTextField5.getText().charAt(i))){
424                     segedpeldany.hibasBevitelFigyelmeztetes(jTextField5, "Helytelen házszám!"); 
425                     break;
426                 }
427             }
428         }
429         if(hibasadat == 0){
430             if(jTextField6.getText().isEmpty()){
431                 segedpeldany.hibasBevitelFigyelmeztetes(jTextField6, "Nincs megadva cégnév!");
432             }
433         }
434         /*koordináták ellenorzese*/
435         if(hibasadat == 0){
436             String[] tomb1 = jTextField7.getText().split("\\.");
437             segedpeldany.koordinataValidator(jTextField7, tomb1, 1);
438            
439         }/*2.koord*/
440         if(hibasadat == 0){
441             String[] tomb2 = jTextField8.getText().split("\\.");
442             segedpeldany.koordinataValidator(jTextField8, tomb2, 2);
443         
444         }
445         /*ha minden adat jo, felvesszuk, flag es mezok nullazasa*/

728      * Koordináta formátumának ellenőrzésére használt metódus.<br>
729      * Ha rossz a formátum, akkor egy MessageDialog figyelmeztet az aktuális hibára. Lehetséges hibák: betűt tartalmaz, helytelen formátum, valótlan értéket tartalmaz.
730      * 
731      * @param mezo A mező aminek a tartalmát vizsgáljuk.
732      * @param tomb Az {@code xx.xx.xx,xx} formátumú koordináta "."-ra való szétbontásnál eredményül kapott részsztringeket tartalmazó tömb.
733      * @param k Egy egész szám. Azt jelzi, hogy a cég melyik koordinátájával van baj.
734      */
735     public void koordinataValidator(javax.swing.JTextField mezo, String[] tomb, int k){
736         if(tomb.length != 3){/*szogmp-ben vesszo helyett pont irasara kilép*/
737                     segedpeldany.hibasBevitelFigyelmeztetes(mezo, "Helytelen a(z) "+ k +". koordináta formátuma! \nA szögmásodpercben a . helyett a , az elvárt!");
738                 }
739         if(hibasadat == 0){
740             for(int i=0; i<3; i++){
741                     if(hibasadat == 0){
742                         for(int j=0; j<tomb[i].length(); j++){
743                             if((j==0 && i==0 && tomb[i].charAt(j)=='-') || (i == 2 && tomb[i].charAt(j)==',')){
744                                 j++;
745                             }
746                             if( ! Character.isDigit(tomb[i].charAt(j))){
747                                 segedpeldany.hibasBevitelFigyelmeztetes(mezo, "Helytelen a(z) "+ k +". koordináta formátuma! Betűt tartalmaz!");
748                                 break;
749                             }
750                         }
751                     }
752                     if(hibasadat == 0 && i==0){
753                         if((k == 1) && (Integer.parseInt(tomb[0]) > 90 || Integer.parseInt(tomb[0]) < -90)){
754                             segedpeldany.hibasBevitelFigyelmeztetes(mezo, "Az első koordináta nem lehet 90 foktól nagyobb, illetve -90 foktól kisebb!");
755                             break;
756                         }
757                     }
758                     if(hibasadat == 0 && i==0){
759                         if((k == 2) && (Integer.parseInt(tomb[0]) > 180 || Integer.parseInt(tomb[0]) < -180)){
760                             segedpeldany.hibasBevitelFigyelmeztetes(mezo, "A második koordináta nem lehet 180 foktól nagyobb, illetve -180 foktól kisebb!");
761                             break;
762                         }  
763                     }
764                     if(hibasadat == 0 && i==1){
765                         if(Integer.parseInt(tomb[1]) > 60 || Integer.parseInt(tomb[1]) < 0){
766                             segedpeldany.hibasBevitelFigyelmeztetes(mezo, "A szögperc nem lehet 60-nál nagyobb, illetve 0-nál kisebb!");
767                             break;
768                         }
769                     }
770                     if(hibasadat == 0 && i==2){
771                         String tmp = tomb[2].replace(',', '.');
772                         if(Double.parseDouble(tmp) > 60.0 || Double.parseDouble(tmp) < 0.0){
773                             segedpeldany.hibasBevitelFigyelmeztetes(mezo, "A szögmásodperc nem lehet 60-nál nagyobb, illyetve 0-nál kisebb!");
774                             break;
775                         }

778                     }
